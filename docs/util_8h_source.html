100 static inline void qryptext_dev_urandom(uint8_t* output_buffer, const size_t output_buffer_size)
101 {
102  if (output_buffer != NULL && output_buffer_size > 0)
103  {
104 #ifdef _WIN32
105  BCryptGenRandom(NULL, output_buffer, output_buffer_size, BCRYPT_USE_SYSTEM_PREFERRED_RNG);
106 #else
107  FILE* rnd = fopen("/dev/urandom", "r");
108  if (rnd != NULL)
109  {
110  fread(output_buffer, sizeof(uint8_t), output_buffer_size, rnd);
111  fclose(rnd);
112  }
113 #endif
114  }
115 }
